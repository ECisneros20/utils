def setup_logging() -> None:
    """Setea el mensaje de logging para todos los scripts del pj_utils y también del
    proyecto
    """
    # Ruta del archivo log
    archivo_log = f"{Constantes.ruta_log.value}/app.log"
    # Configuración del logger
    logging.basicConfig(
        format="%(asctime)s - %(name)s - %(levelname)s - %(message)s",
        datefmt="%Y-%m-%d %H:%M:%S",
        level=logging.INFO,
        handlers=[
            logging.FileHandler(
                filename=archivo_log, mode="a", encoding=Constantes.encoding.value
            )
        ],
    )

def agregar_log_bd(mensaje: str, credenciales: dict = {}) -> None:
    """Copia el último mensaje del archivo log y lo agrega a la tabla log de PostgreSQL.
    Para cada nuevo proyecto es necesario cambiar las variables tabla y columnas, según
    las credenciales que se usarán aparte de las siguientes:

        'marca_tiempo - modulo - nivel - mensaje'

    Args:
        mensaje (str): _description_
        credenciales (dict, optional): Datos a registrar en el log. Defaults to {}.
    """
    # Configura el logging
    setup_logging()
    # Obtiene un logger para este módulo
    logger = logging.getLogger(__name__)
    # Intenta agregar datos del log a la bd
    try:
        if isinstance(mensaje, str) and isinstance(credenciales, dict):
            # Datos a modificar para cada proyecto (tabla, columnas)
            tabla = "esq_transcriptor_flagrancia.tbl_log_transcriptor_flagrancia"
            columnas = "marca_tiempo, modulo, nivel, mensaje"
            reemplazos = ", ".join(["%s"] * len(columnas.split(",")))
            query = f"INSERT INTO {tabla} ({columnas}) VALUES ({reemplazos})"
            # Extraer el mensaje más reciente del archivo .log
            archivo_log = f"{Constantes.ruta_log.value}/app.log"
            patron = r"\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}"
            with open(archivo_log, "r", encoding=Constantes.encoding.value) as archivo:
                log = archivo.readlines()
                log = "".join(log)
                if match := re.finditer(patron, log):
                    pos_inicial = list(match)[-1].start()
                pos_final = log.rfind(mensaje)
                log = log[pos_inicial : pos_final + len(mensaje)].strip()
                datos_log = log.split(" - ")
            # Establecer conexión con base de datos
            cred_acceso = {
                "dbname": Constantes.pg_dbname.value,
                "user": Constantes.pg_user.value,
                "password": Constantes.pg_password.value,
                "host": Constantes.pg_host.value,
                "port": Constantes.pg_port.value,
            }
            with psycopg.connect(**cred_acceso) as conn:
                # Agregar credenciales y subir toda la información en la tabla de log
                if len(credenciales) > 0:
                    datos_log.extend(credenciales.values())
                if len(datos_log) == len(columnas.split(",")):
                    conn.execute(query, datos_log)
                    conn.commit()
                else:
                    msj = f"No coinciden las dimensiones: {datos_log} vs {columnas}"
                    logger.error(msj)
        else:
            msj = f"{type(mensaje)} y {type(credenciales)} no son 'str' y 'dict'"
            logger.exception(msj)
    except Exception as e:
        msj = f"Error al intentar guardar el log en BD: {e}"
        logger.exception(msj)
